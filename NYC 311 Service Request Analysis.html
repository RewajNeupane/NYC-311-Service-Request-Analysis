from scipy.stats import f_oneway

# Create df_clean by removing rows with missing or negative Request_Closing_Time
df_clean = df.dropna(subset=['Request_Closing_Time', 'Complaint Type']).copy()

# Convert 'Request_Closing_Time' to numeric (in hours) and filter out rows with negative or zero values
df_clean['Request_Closing_Time'] = df_clean['Request_Closing_Time'].dt.total_seconds() / 3600  # Convert to hours
df_clean = df_clean[df_clean['Request_Closing_Time'] > 0]

# Filter out complaint types with too few samples for statistical validity (e.g., at least 30 entries)
complaint_counts = df_clean['Complaint Type'].value_counts()
valid_complaints = complaint_counts[complaint_counts > 30].index
filtered_df = df_clean[df_clean['Complaint Type'].isin(valid_complaints)]

# Prepare data for ANOVA test: list of arrays of request closing times grouped by complaint type
groups = [group["Request_Closing_Time"].values for name, group in filtered_df.groupby("Complaint Type")]

# Perform One-Way ANOVA
f_stat, p_value = f_oneway(*groups)

# Print results
print("F-statistic:", f_stat)
print("p-value:", p_value)

# Interpret the result based on p-value
if p_value < 0.05:
    print("→ Reject the null hypothesis")
else:
    print("→ Fail to reject the null hypothesis")

from scipy.stats import chi2_contingency


# Remove rows with missing Complaint Type or Borough
df_clean = df.dropna(subset=['Complaint Type', 'Borough'])

# Create a contingency table of complaint types by borough
table = pd.crosstab(df_clean['Complaint Type'], df_clean['Borough'])

# Perform Chi-Square test
chi2, p, dof, expected = chi2_contingency(table)

# Print results
print(f"Chi-square: {chi2:.4f}")
print(f"P-value: {p:.6f}")

# Conclusion
if p < 0.05:
    print("Reject the null hypothesis: There is a significant relationship between complaint type and location")
else:
